3. 干掉字符串中出现最多的字符

```js
/*
*干掉字符串中出现最多的字符
*/
let s = 'asdfasfjgijjjjjadsf'
let map = new Map()
let maxCount = 0
let maxCountCh = ''
for (let i = 0; i < s.length; i++) {
  let count = map.get(s[i]) || 0
  map.set(s[i], count + 1)
  if (count + 1 > maxCount) {
    maxCount = count + 1
    maxCountCh = s[i]
  }
}
// node环境下竟然没有replaceAll
let t = ''
for(let i = 0; i < s.length; i++){
  if(s[i] === maxCountCh) continue
  else t = t + s[i]
}
console.log(t)
```
